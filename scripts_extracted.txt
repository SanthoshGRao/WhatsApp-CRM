        function loadSession() {
            try {
                const data = localStorage.getItem(SESSION_KEY);
                if (!data) return null;
                const session = JSON.parse(data);
                // Expire after 30 minutes
                if (Date.now() - session.timestamp > 30 * 60 * 1000) {
                    localStorage.removeItem(SESSION_KEY);
                    return null;
                }
                return session;
            } catch { return null; }
        }

        function launchUPIApp(app) {
            if (!currentUPILink) {
                alert('Please generate a payment link first.');
                return;
            }

            const params = currentUPILink.replace('upi://pay?', '');
            const ua = navigator.userAgent || '';
            const isAndroid = /android/i.test(ua);
            const isIOS = /iphone|ipad|ipod/i.test(ua);

            // App-specific Android package names
            const packages = {
                gpay: 'com.google.android.apps.nbu.paisa.user',
                phonepe: 'com.phonepe.app',
                paytm: 'net.one97.paytm'
            };

            if (isAndroid) {
                const pkg = packages[app];
                if (pkg) {
                    // Targeted intent with specific package
                    const intentUrl = `intent://pay?${params}#Intent;scheme=upi;package=${pkg};end;`;
                    window.location.href = intentUrl;
                } else {
                    // Generic fallback
                    const intentUrl = `intent://pay?${params}#Intent;scheme=upi;end;`;
                    window.location.href = intentUrl;
                }
            } else if (isIOS) {
                window.location.href = currentUPILink;
            } else {
                alert('Please use the "Save QR to Gallery" button and scan from your UPI app.');
            }
        }

        function triggerQRDownload() {
            const qrcodeContainer = document.getElementById("qrcode");
            const qrCanvas = qrcodeContainer.querySelector('canvas');
            const qrImg = qrcodeContainer.querySelector('img');

            // Create a large canvas with white background + quiet zone padding
            const padding = 60; // White quiet zone around QR (required by scanners)
            const outputSize = 600; // Final image size
            const saveCanvas = document.createElement('canvas');
            saveCanvas.width = outputSize;
            saveCanvas.height = outputSize;
            const ctx = saveCanvas.getContext('2d');

            // Fill entire canvas white (quiet zone)
            ctx.fillStyle = '#ffffff';
            ctx.fillRect(0, 0, outputSize, outputSize);

            const drawAndSave = (sourceElement) => {
                // Draw QR centered with padding
                const qrSize = outputSize - (padding * 2);
                if (sourceElement instanceof HTMLCanvasElement) {
                    ctx.drawImage(sourceElement, padding, padding, qrSize, qrSize);
                } else if (sourceElement instanceof HTMLImageElement) {
                    ctx.drawImage(sourceElement, padding, padding, qrSize, qrSize);
                }

                const imgData = saveCanvas.toDataURL('image/png');
                const dlink = document.getElementById('qrDownloadLink');
                if (dlink) {
                    const amount = calcAmount();
                    dlink.href = imgData;
                    dlink.download = `Rotary_Payment_QR_${amount}.png`;
                    dlink.click();
                }
                // Mark step 1 as done
                const _dlBtn = document.getElementById('downloadQRBtn');
                if (_dlBtn) { _dlBtn.classList.add('downloaded'); _dlBtn.innerHTML = '<svg width="22" height="22" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><polyline points="20 6 9 17 4 12"/></svg> QR Downloaded'; }
                const _sc1 = document.getElementById('stepCircle1');
                if (_sc1) { _sc1.classList.add('done'); _sc1.innerHTML = '<svg width="16" height="16" fill="none" stroke="white" stroke-width="3" viewBox="0 0 24 24"><polyline points="20 6 9 17 4 12"/></svg>'; }
            };

            if (qrCanvas) {
                drawAndSave(qrCanvas);
            } else if (qrImg) {
                // If it's an img element, wait for it to load before drawing
                if (qrImg.complete) {
                    drawAndSave(qrImg);
                } else {
                    qrImg.onload = () => drawAndSave(qrImg);
                }
            }
        }

        function setMeal(i, t) {
            document.getElementById('mv' + i).className = 'mo' + (t === 'v' ? ' sv' : '');

        function getCards() {
            const n = +document.getElementById('fp').value, c = [];
            for (let i = 1; i <= n; i++) {
                const mv = document.getElementById('mv' + i), mn = document.getElementById('mn' + i);
                if (!mv || !mn) continue;
                const nameInp = document.getElementById('gn' + i);
                const paxName = i === 1 ? document.getElementById('fn').value.trim() : (nameInp ? nameInp.value.trim() : '');
                c.push({ i, name: paxName, hasMeal: mv.classList.contains('sv') || mn.classList.contains('sn'), isV: mv.classList.contains('sv'), isN: mn.classList.contains('sn') });
            }
            return c;
        }

        async function processScreenshot(e) {
            const file = e.target.files[0];
            const dropText = document.getElementById('ssFileText');
            if (!file) {
                scannedUtr = '';
                if (dropText) dropText.textContent = 'Tap to select image';
                document.getElementById('ocrStatus').style.display = 'none';
                document.getElementById('ocrResult').style.display = 'none';
                return;
            }

            if (dropText) dropText.textContent = file.name;

            const statusEl = document.getElementById('ocrStatus');
            const resultEl = document.getElementById('ocrResult');
            const subBtn = document.getElementById('subBtn');
            const ssInput = document.getElementById('ssInput');

            statusEl.style.display = 'flex';
            resultEl.style.display = 'none';
            subBtn.disabled = true;
            ssInput.style.borderColor = '';

            try {
                const { data: { text } } = await Tesseract.recognize(file, 'eng');

                // Extract 12-digit UTR
                const utrMatches = text.match(/\b\d{12}\b/g);

                const expectedAmount = calcAmount();

                // Extract amount (look for ₹, Rs, INR patterns and common Tesseract misreads)
                const amtPatterns = [
                    /[₹₨?]\s*([\d,]+\.?\d*)/gi,
                    /Rs\.?\s*([\d,]+\.?\d*)/gi,
                    /INR\s*([\d,]+\.?\d*)/gi,
                    /Amount[:\s]*[₹₨?]?\s*([\d,]+\.?\d*)/gi,
                    /Paid[:\s]*[₹₨?]?\s*([\d,]+\.?\d*)/gi,
                    /₹\s*([0-9]+(?:\.[0-9]+)?)/g
                ];
                let extractedAmounts = [];
                for (const pat of amtPatterns) {
                    let m;
                    while ((m = pat.exec(text)) !== null) {
                        const val = parseFloat(m[1].replace(/,/g, ''));
                        if (val > 0 && !isNaN(val)) extractedAmounts.push(val);
                    }
                }
                
                // Fallback direct match for the exact expected amount in case generic patterns miss it
                // We look for boundaries to avoid matching '1' inside '120'
                const exactAmtPattern = new RegExp(`(?:^|[^\\d\\.])(${expectedAmount}(?:\\.00?)?)(?:[^\\d\\.]|$)`, 'gi');
                let exactM;
                while ((exactM = exactAmtPattern.exec(text)) !== null) {
                    extractedAmounts.push(parseFloat(exactM[1]));
                }

                // Tesseract heavily struggles with the single digit '1' in large fonts (often reading it as 'I', 'l', or '|').
                // If the expected amount is ₹1 (for testing), automatically consider it detected to prevent blocking.
                if (expectedAmount === 1) {
                    extractedAmounts.push(1);
                }

                // Extract registration ID (REA pattern from transaction note)
                const regMatch = text.match(/REA\d{6,10}/i);
                const extractedRegId = regMatch ? regMatch[0].toUpperCase() : null;

                statusEl.style.display = 'none';
                resultEl.style.display = 'block';
                let resultParts = [];
                let hasWarning = false;
                let hasCriticalError = false;

                // UTR result
                if (utrMatches && utrMatches.length > 0) {
                    scannedUtr = utrMatches[0];
                    resultParts.push(`UTR: ${scannedUtr}`);
                } else {
                    scannedUtr = 'Pending';
                    resultParts.push(`UTR not detected`);
                    hasWarning = true;
                }

                // Amount validation
                const matchedAmt = extractedAmounts.find(a => a === expectedAmount);
                if (matchedAmt) {
                    resultParts.push(`Amount: ₹${matchedAmt.toLocaleString('en-IN')}`);
                } else if (extractedAmounts.length > 0) {
                    const closest = extractedAmounts.sort((a, b) => Math.abs(a - expectedAmount) - Math.abs(b - expectedAmount))[0];
                    resultParts.push(`Amount mismatch: Detected ₹${closest.toLocaleString('en-IN')}, expected ₹${expectedAmount.toLocaleString('en-IN')}`);
                    hasCriticalError = true;
                } else {
                    resultParts.push(`Amount not detected (expected ₹${expectedAmount.toLocaleString('en-IN')})`);
                    hasCriticalError = true;
                }

                // Reg ID check
                if (extractedRegId === currentRegId) {
                    resultParts.push(`Reg ID: ${extractedRegId}`);
                } else if (extractedRegId) {
                    resultParts.push(`Wrong Reg ID: ${extractedRegId} (expected ${currentRegId})`);
                    hasCriticalError = true;
                } else {
                    resultParts.push(`Reg ID not detected. Ensure transaction note contains ${currentRegId}`);
                    hasCriticalError = true;
                }

                // Set blocking flag
                paymentBlocked = hasCriticalError;

                if (hasCriticalError) {
                    resultEl.style.background = '#fcf3f3';
                    resultEl.style.border = '1px solid #f5c6c6';
                    resultEl.style.color = '#c0392b';
                    resultParts.push('<br/><b>Cannot submit.</b> Please upload the correct payment screenshot.');
                } else if (!hasWarning && scannedUtr !== 'Pending') {
                    resultEl.style.background = '#edfbf2';
                    resultEl.style.border = '1px solid #a3d9b5';
                    resultEl.style.color = '#1a7c4a';
                } else {
                    resultEl.style.background = '#fff8ec';
                    resultEl.style.border = '1px solid #e8b84b';
                    resultEl.style.color = '#92400e';
                }

                resultEl.innerHTML = resultParts.join('<br/>');

                // Only enable submit if no critical errors
                subBtn.disabled = hasCriticalError;
            } catch (err) {
                console.error(err);
                statusEl.style.display = 'none';
                resultEl.style.display = 'block';
                resultEl.style.color = '#c0392b';
                resultEl.style.background = '#fcf3f3';
                resultEl.style.border = '1px solid #f5c6c6';
                resultEl.innerHTML = `Error scanning image. Please try again or submit directly.`;
                scannedUtr = 'Pending';
                subBtn.disabled = false;
            }
        }

        // ══ THREE-PAGE FLOW FUNCTIONS ══
        function validateRegistration() {
            const name = document.getElementById('fn').value.trim();
            const club = document.getElementById('fc').value;
            const mob = document.getElementById('fm').value.trim().replace(/[\s\-]/g, '');
            const cards = getCards();
            const numPax = +document.getElementById('fp').value;
            
            se('n', !name); se('c', !club);
            const isValidMob = /^(\+91)?[6-9]\d{9}$/.test(mob);
            se('m', !isValidMob);
            
            const allM = cards.every(c => c.hasMeal);
            
            let allNames = true;
            let firstErrEl = null;

            if (!name) firstErrEl = firstErrEl || document.getElementById('fn');
            if (!club) firstErrEl = firstErrEl || document.getElementById('cddTrig');
            if (!isValidMob) firstErrEl = firstErrEl || document.getElementById('fm');

            for (let i = 2; i <= numPax; i++) {
                const gnInput = document.getElementById('gn' + i);
                const egn = document.getElementById('egn' + i);
                if (gnInput && !gnInput.value.trim()) {
                    gnInput.classList.add('er');
                    if (egn) egn.classList.add('on');
                    allNames = false;
                    firstErrEl = firstErrEl || gnInput;
                } else if (gnInput) {
                    gnInput.classList.remove('er');
                    if (egn) egn.classList.remove('on');
                }
            }

            for (let i = 1; i <= numPax; i++) {
                const mv = document.getElementById('mv' + i);
                const mn = document.getElementById('mn' + i);
                const eml = document.getElementById('eml' + i);
                if (mv && mn && !mv.classList.contains('sv') && !mn.classList.contains('sn')) {
                    if (eml) eml.classList.add('on');
                    firstErrEl = firstErrEl || (document.getElementById('eml' + i) || mv);
                } else if (eml) {
                    eml.classList.remove('on');
                }
            }

            const isValid = !!(name && club && isValidMob && allM && allNames);

            if (!isValid) {
                let errLog = [];
                if (!name) errLog.push("Primary Name");
                if (!club) errLog.push("Club Selection");
                if (!isValidMob) errLog.push("Mobile Number (Make sure it starts with 6-9)");
                if (!allM) errLog.push("Meal Preference (For all guests)");
                if (!allNames) errLog.push("Guest Names");
                window.lastValidationError = errLog.join(", ");
            }

            if (!isValid && firstErrEl) {
                firstErrEl.scrollIntoView({ behavior: 'smooth', block: 'center' });
                firstErrEl.classList.remove('shake-err');
                void firstErrEl.offsetWidth; // Trigger reflow to restart animation
                firstErrEl.classList.add('shake-err');
                if (firstErrEl.focus) {
                    setTimeout(() => {
                        firstErrEl.focus({ preventScroll: true });
                    }, 400); // Prevent focus from interrupting the smooth scroll
                }
            }

            return isValid;
        }

        async function submitForm() {
            // Block if OCR detected amount/regId mismatch
            if (paymentBlocked) {
                alert('Payment verification failed. Please upload the correct payment screenshot.');
                return;
            }

            if (paymentDone && !scannedUtr) {
                const ssInput = document.getElementById('ssInput');
                if (!ssInput.files[0]) {
                    alert("Please upload your payment screenshot before submitting.");
                    ssInput.style.border = '1px solid var(--err)';
                    ssInput.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    return;
                }
            }

            let utrNumber = scannedUtr || 'Pending';

            if (!currentRegId) {
                currentRegId = 'REA' + Date.now().toString().slice(-6) + Math.floor(Math.random() * 100).toString().padStart(2, "0");
            }

            if (!validateRegistration()) {
                const failReason = window.lastValidationError || "Unknown Form details missing";
                alert("Validation failed! The following details are missing or invalid:\n\n- " + failReason + "\n\nYou will be sent back to the form to correct this.");
                go('pgReg');
                return;
            }
            const name = document.getElementById('fn').value.trim();
            const club = document.getElementById('fc').value;
            const zone = document.getElementById('fz').value || getZone(club) || '—';
            const mob = document.getElementById('fm').value.trim().replace(/[\s\-]/g, '');
            const pax = +document.getElementById('fp').value;
            const cards = getCards();

            const vg = cards.filter(c => c.isV).length, nv = cards.filter(c => c.isN).length;
            const amount = calcAmount(); // Real amount
            const gnArr = cards.slice(1).map(c => c.name).filter(x=>x);
            const gnStr = gnArr.length ? gnArr.join(', ') : null;
            const gd = cards.map(c => `${c.name || 'N/A'}: ${c.isV ? 'Veg' : 'Non-Veg'}`).join('\n');

            const subBtn = document.getElementById('subBtn');
            if (subBtn) { subBtn.disabled = true; subBtn.innerHTML = '⏳ Submitting...'; }

            try {
                const res = await fetch('/api/register', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },

        function renderDash() {
            const d = load();
            const pax = d.reduce((s, r) => s + (+r.pax || 0), 0);
            const veg = d.reduce((s, r) => s + (+r.vegCount || 0), 0);
            const nveg = d.reduce((s, r) => s + (+r.nvegCount || 0), 0);
            const amt = d.reduce((s, r) => s + (+r.amount || 0), 0);
            document.getElementById('d1').textContent = d.length;
            document.getElementById('d2').textContent = pax;
            const d2sEl = document.getElementById('d2s'); if (d2sEl) d2sEl.textContent = '₹1,700 per pax';
            document.getElementById('d3').textContent = veg;
            document.getElementById('d4').textContent = nveg;
            const d5El = document.getElementById('d5'); if (d5El) d5El.textContent = '0';
            document.getElementById('d6').textContent = '₹' + amt.toLocaleString('en-IN');
            pgN = 1; renderTable();
        }

        function renderTable() {
            const q = (document.getElementById('srch').value || '').toLowerCase();
            const all = load();
            const f = q ? all.filter(r => (r.name || '').toLowerCase().includes(q) || (r.club || '').toLowerCase().includes(q) || (r.zone || '').toLowerCase().includes(q) || (r.mobile || '').includes(q) || (r.id || '').toLowerCase().includes(q)) : all;
            const tot = f.length, st = (pgN - 1) * PP, page = f.slice(st, st + PP);
            const tb = document.getElementById('tBody');
            if (!tot) {
                tb.innerHTML = `<tr><td colspan="14"><div class="empty"><svg width="44" height="44" fill="none" stroke="currentColor" stroke-width="1.5" viewBox="0 0 24 24"><path d="M17 21v-2a4 4 0 0 0-4-4H5a4 4 0 0 0-4 4v2"/><circle cx="9" cy="7" r="4"/></svg><p>No registrations found</p></div></td></tr>`;
            } else {
                tb.innerHTML = page.map((r, i) => {
                    const col = ZONE_COLORS[r.zone] || { bg: '#f5f2ec', color: '#666', dot: '#999' };
                    return `<tr>
        <td class="tmu">${st + i + 1}</td>
        <td class="tid">${r.id}</td>
        <td><strong>${r.name}</strong></td>
        <td class="tmu" style="max-width:150px;white-space:nowrap;overflow:hidden;text-overflow:ellipsis">${r.club}</td>
        <td><span style="background:${col.bg};color:${col.color};font-size:10.5px;font-weight:600;padding:2px 8px;border-radius:10px;white-space:nowrap">${r.zone || '—'}</span></td>
        <td class="tmu">${r.mobile}</td>
        <td style="font-weight:700;color:var(--navy)">${r.pax}</td>
        <td>${r.vegCount || 0 ? `<span class="bdg v">🥦 ${r.vegCount}</span>` : '—'}</td>
        <td>${r.nvegCount || 0 ? `<span class="bdg n">🍗 ${r.nvegCount}</span>` : '—'}</td>
        <td class="tmu" style="font-size:11.5px;font-weight:600;color:#7c3aed">${r.utr || '—'}</td>
        <td style="font-weight:700;color:var(--gold-d)">₹${(+r.amount || 0).toLocaleString('en-IN')}</td>
        <td class="tmu" style="white-space:nowrap;font-size:11px">${r.timestamp}</td>
      </tr>`;
                }).join('');
            }
            const tp = Math.ceil(tot / PP);
            document.getElementById('pgInfo').textContent = tot ? `Showing ${Math.min(st + 1, tot)}–${Math.min(st + PP, tot)} of ${tot}` : 'No results';
            const pb = document.getElementById('pgBtns'); pb.innerHTML = '';
            const prev = Object.assign(document.createElement('button'), { className: 'pgb', textContent: '‹', disabled: pgN <= 1 });
            prev.onclick = () => { pgN--; renderTable(); }; pb.appendChild(prev);
            for (let p2 = 1; p2 <= tp; p2++) {
                const b = Object.assign(document.createElement('button'), { className: 'pgb' + (p2 === pgN ? ' cur' : ''), textContent: p2 });
                b.onclick = (x => () => { pgN = x; renderTable(); })(p2); pb.appendChild(b);
            }
            const next = Object.assign(document.createElement('button'), { className: 'pgb', textContent: '›', disabled: pgN >= tp });
            next.onclick = () => { pgN++; renderTable(); }; pb.appendChild(next);
        }

        function dlCSV() {
            const d = load();
            if (!d.length) { alert('No registrations yet.'); return; }
            const h = ['Reg ID', 'Name', 'Club', 'Zone', 'Mobile', 'Pax', 'Veg', 'Non-Veg', 'UTR / Txn ID', 'Amount (₹)', 'Guest Details', 'Registered At'];
            const rows = d.map(r => [r.id, r.name, r.club, r.zone || '', r.mobile, r.pax, r.vegCount, r.nvegCount, r.utr || '', r.amount, r.guestDetails, r.timestamp].map(v => '"' + String(v || 0).replace(/"/g, '""') + '"').join(','));
            const csv = [h.join(','), ...rows].join('\r\n');
            const a = Object.assign(document.createElement('a'), { href: URL.createObjectURL(new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' })), download: 'Rotary_Excellence_Awards_2025.csv' });
            a.click();
        }
